
        func quicksort(arr []int) []int {
            if len(arr) < 2 {
                return arr;
            }
            pivot := arr[0];
            var less, greater []int;
            for _, val := range arr[1:] {
                if val <= pivot {
                    less = append(less, val);
                } else {
                    greater = append(greater, val);
                }
            }
            return append(append(quicksort(less), pivot), quicksort(greater));
        }
